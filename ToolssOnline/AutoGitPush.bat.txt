@echo off
REM === مسیر پروژه خودت رو اینجا وارد کن ===
cd /d "C:\Users\hp\source\repos\Toolss\ToolssOnline"

REM === زمان فعلی برای پیام کامیت ===
for /f "tokens=1-4 delims=/ " %%a in ('date /t') do (
    set DATE=%%a-%%b-%%c
)
for /f "tokens=1-2 delims=: " %%a in ('time /t') do (
    set TIME=%%a-%%b
)

git add .
git commit -m "Auto backup %DATE% %TIME%"
git push

echo.
echo ===== Push done! =====
pause
